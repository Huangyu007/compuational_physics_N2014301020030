def trans(t):
    s = [list(t[i]) for i in range(len(t))]
    return s

def output(t):
    for i in range(len(t)):
        print (''.join(t[i]))

def center(t):
    sumx = sumy = x = y = 0
    for i in range(len(t)):
        for j in range(len(t)):
            if t[i][j] != " ":
                sumx=sumx+j
                x=x+1
                sumy=sumy+i
                y=y+1
    return (sumy//y , sumx//x)

def target(a,b,c,t):
    xy = a-center(t)[0]
    xx = b-center(t)[1]
    dis = math.sqrt(xy**2+xx**2)
    if xx != 0 :
        if xx>0 and xy<0:
            ang = math.atan(xy/xx)
        elif xx>0 and xy>0:
            ang = math.atan(xy/xx)
        elif xx<0 and xy>0:
            ang = math.atan(xy/xx)+math.pi
        elif xx<0 and xy<0:
            ang = math.atan(xy/xx)-math.pi
        elif xx>0 and xy==0:
            ang = 0
        elif xx<0 and xy==0:
            ang = -math.pi
    else:
        if xy>0:
            ang = math.pi/2
        else:
            ang = -math.pi/2
    y = dis*math.sin(ang+c)+center(t)[0]
    x = dis*math.cos(ang+c)+center(t)[1]
    return (int(y) , int(x))

def turn(t,c):
    s = [[" "]*len(t) for i in range(len(t))]
